% --- SLIDE 1 ---
\begin{frame}{Introduction \& Objectives}
  \begin{itemize}
    \item \textbf{Mathematical Morphology (MM):} Robust framework for extracting region descriptors (boundaries, skeletons) and noise reduction.
    \item \textbf{The Goal:} Accelerate morphological operations for high-resolution imagery, transitioning from a sequential approach to a highly optimized parallel framework.
    \item \textbf{Technology Stack:} \texttt{C++17} and \texttt{OpenMP}.
    \item \textbf{Benchmark Dataset:} BSDS500 (converted to grayscale, scaled up to 4x to evaluate scalability).
  \end{itemize}
\end{frame}

% --- SLIDE 2 ---
\begin{frame}{Context \& Sequential Baseline}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \begin{block}{Fundamental Operators}
        \begin{itemize}
          \item \textbf{Erosion:} Minimum local value (discards small-scale noise).
          \item \textbf{Dilation:} Maximum local value.
        \end{itemize}
      \end{block}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{colorblock}[white]{sintefred}{The Bottleneck}
        \begin{itemize}
          \item Naive 2D stencil execution scales quadratically: $\mathcal{O}(M \times N \times K^2)$.
          \item \textbf{Compound operations} (Opening/Closing) require multiple full-image passes.
          \item Destroys \textit{temporal locality} and creates severe memory bandwidth issues.
        \end{itemize}
      \end{colorblock}
    \end{column}
  \end{columns}
\end{frame}

% --- SLIDE 3 ---
\begin{frame}{Hardware Approach: OpenMP Parallelization}
  \framesubtitle{Exploiting Embarrassingly Parallel workloads}
  \begin{itemize}
    \item Morphological stencils are \textit{read-only} regarding the input state (no loop-carried dependencies).
    \item \textbf{Loop Collapsing:} \texttt{collapse(2)} merges X and Y spatial loops, increasing work granularity and preventing thread starvation.
    \item \textbf{Memory Safety:} Use of \texttt{default(none)} and strict variable scoping (e.g., local \texttt{min\_val}) avoids locks/atomic synchronizations.
    \item \textbf{Scheduling Strategy:} While workload per pixel is theoretically static, \textbf{dynamic scheduling} marginally outperforms static by mitigating \textit{false sharing} and OS scheduling jitter.
  \end{itemize}
\end{frame}

% --- SLIDE 4 ---
\begin{frame}{Algorithmic Optimization: 1D Separable Kernels}
  \begin{itemize}
    \item \textbf{The Dominant Lever:} Transitioning from a 2D stencil to two consecutive 1D passes (Horizontal $\rightarrow$ Vertical).
    \item \textbf{Complexity Reduction:} From $\mathcal{O}(K^2)$ down to $\mathcal{O}(2K)$.
      \begin{itemize}
        \item \textit{Example:} A $15 \times 15$ kernel drops from 225 to just 30 operations per pixel.
      \end{itemize}
  \end{itemize}
  
  \begin{block}{The Cache Asymmetry Issue}
    \begin{itemize}
      \item \textbf{Horizontal Pass:} Optimal spatial locality. Glides along adjacent columns mirroring row-major memory layout.
      \item \textbf{Vertical Pass:} Severely degraded. Reads vertically across different rows, overwhelming the cache and causing memory-bound slowdowns.
    \end{itemize}
  \end{block}
\end{frame}

% --- SLIDE 5 ---
\begin{frame}{Architecture: Producer-Consumer Pipeline}
  \framesubtitle{Solving the Fork-Join problem for compound operations}
  \begin{itemize}
    \item \textbf{The Challenge:} 1D Opening requires 4 strict passes (H-Erosion $\rightarrow$ V-Erosion $\rightarrow$ H-Dilation $\rightarrow$ V-Dilation). A standard Fork-Join doubles memory footprint and synchronization barriers.
    \item \textbf{The Pipeline Solution:}
      \begin{itemize}
        \item Thread pool is halved within a single parallel region.
        \item \textbf{Producers (Erosion):} Compute 1D passes and store in an intermediate buffer.
        \item \textbf{Consumers (Dilation):} Fetch immediately from the buffer to compute final passes.
      \end{itemize}
    \item \textbf{Result:} Overlaps execution, drastically reduces global memory traffic, and improves temporal locality.
  \end{itemize}
\end{frame}

% --- SLIDE 6 ---
\begin{frame}{Experimental Setup}
  \begin{itemize}
    \item \textbf{Environment:} Benchmarks conducted on a physical quad-core CPU architecture to strictly trace hardware scaling limits.
    \item \textbf{Dataset (BSDS500):} Diverse real-world image benchmark.
    \item \textbf{Scaling Methodology:} 
      \begin{itemize}
        \item \textit{Strong Scaling:} Evaluating performance gains while increasing thread count on fixed image sizes.
        \item \textit{Weak Scaling:} Images upscaled by 2x and 4x to analyze throughput when proportional workload is added.
      \end{itemize}
  \end{itemize}
\end{frame}

% --- SLIDE 10 ---
\begin{frame}{Ineffective Strategies \& Lessons Learned}
  \framesubtitle{What didn't work and why}
  \begin{itemize}
    \item \textbf{Explicit SIMD Annotations:} 
      \begin{itemize}
        \item \texttt{\#pragma omp simd} proved ineffective in accelerating the core loops. 
        \item This is likely due to the compiler's auto-vectorization already being optimal, or the \texttt{min}/\texttt{max} conditional branching limiting SIMD register exploitation.
      \end{itemize}
    \vspace{0.3cm}
    \item \textbf{Column-Major Traversal:} 
      \begin{itemize}
        \item Attempting to traverse memory column-by-column to speed up the memory-bound Vertical 1D pass failed.
        \item RAM physical architecture is inherently \textit{row-major}; software workarounds introduced unnecessary overhead without yielding any cache benefits.
      \end{itemize}
  \end{itemize}
\end{frame}

% --- SLIDE 11 ---
\begin{frame}{Peak Performance \& Conclusions}
  \begin{columns}
    \begin{column}{0.45\textwidth}
      \begin{colorblock}[white]{sintefblu1}{Optimal Configuration}
        \begin{center}
          \Large \textbf{22x Peak Speedup}
        \end{center}
        Achieved combining 1D kernels and OpenMP ($K=15$, 4 threads) against the sequential baseline.
      \end{colorblock}
    \end{column}
    \begin{column}{0.5\textwidth}
      \begin{itemize}
        \item \textbf{Core Takeaway:} True acceleration requires a \textit{multi-tiered} approach. Hardware scaling a flawed $\mathcal{O}(K^2)$ algorithm is insufficient without algorithmic complexity reduction.
        \item \textbf{Future Developments:} 
          \begin{itemize}
            \item Extend the framework to Heterogeneous Computing (GPUs via \textbf{CUDA}).
            \item The SIMT architecture and Shared Memory could definitively solve the vertical pass memory bottleneck.
          \end{itemize}
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
